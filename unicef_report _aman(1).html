import polars as pl
import pandas as pd
import geopandas as gpd
import numpy as np
from plotnine import (
    ggplot, aes,
    geom_map, geom_bar, geom_point, geom_smooth, geom_line,
    scale_fill_gradientn, scale_color_manual,
    scale_x_log10, scale_x_continuous,
    coord_flip, coord_equal,
    theme_void, theme_minimal, theme,
    element_text, element_blank, element_rect,
    labs
)
import warnings
warnings.filterwarnings("ignore")

print("✅ All packages loaded successfully.")

# ── Load with Polars ──────────────────────────────────────────────────────
ind1 = pl.read_csv("unicef_indicator_1.csv")
ind2 = pl.read_csv("unicef_indicator_2.csv")
meta = pl.read_csv("unicef_metadata.csv")

# Rename long metadata column names to clean short ones
meta = meta.rename({
    "Population, total":                       "population",
    "GDP per capita (constant 2015 US$)":      "gdp_per_capita",
    "Life expectancy at birth, total (years)": "life_expectancy",
    "Inflation, consumer prices (annual %)":   "inflation",
    "GDP growth (annual %)":                   "gdp_growth"
})

print(f"Indicator 1 — Child Deprivation : {ind1.shape[0]:,} rows, {ind1.shape[1]} columns")
print(f"Indicator 2 — Zinc Treatment    : {ind2.shape[0]:,} rows, {ind2.shape[1]} columns")
print(f"Metadata    — Country Economics : {meta.shape[0]:,} rows, {meta.shape[1]} columns")


# ── Indicator 1: stored as proportions (0–1), convert to % for display ───
# e.g. 0.0666 → 6.66 %
dep_stats = (
    ind1
    .filter(pl.col("sex") == "Total")
    .with_columns((pl.col("obs_value") * 100).alias("obs_pct"))
    .select("obs_pct")
    .describe()
)
print("=== Indicator 1: Child Deprivation (%, after ×100 conversion) ===")
print(dep_stats)
print(f"Countries: {ind1['country'].n_unique()} | "
      f"Years: {ind1['time_period'].min()}–{ind1['time_period'].max()}\n")

# ── Indicator 2: already in percentage scale (0–100) ─────────────────────
zinc_stats = (
    ind2
    .filter(pl.col("sex") == "Total")
    .select("obs_value")
    .describe()
)
print("=== Indicator 2: Zinc Treatment Coverage (%) ===")
print(zinc_stats)
print(f"Countries: {ind2['country'].n_unique()} | "
      f"Years: {ind2['time_period'].min()}–{ind2['time_period'].max()}\n")

# ── Metadata: null check on key economic columns ─────────────────────────
null_check = (
    meta
    .select(["gdp_per_capita", "life_expectancy", "population"])
    .null_count()
)
print("=== Metadata: Null Count in Key Columns ===")
print(null_check)


# ── 1. Average zinc coverage per country (Total sex only) ────────────────
zinc_avg = (
    ind2
    .filter(pl.col("sex") == "Total")
    .group_by(["alpha_3_code", "country"])
    .agg(pl.col("obs_value").mean().alias("zinc_avg"))
)

# ── 2. Load world geometries ──────────────────────────────────────────────
# geopandas built-in works for geopandas < 1.0;
# GitHub GeoJSON fallback covers geopandas >= 1.0
_NE_URL = (
    "https://raw.githubusercontent.com/nvkelso/natural-earth-vector"
    "/master/geojson/ne_110m_admin_0_countries.geojson"
)
try:
    world_gpd = gpd.read_file(gpd.datasets.get_path("naturalearth_lowres"))
except Exception:
    world_gpd = gpd.read_file(_NE_URL)

# Normalise the ISO-A3 column name to 'iso_a3' regardless of source
_raw_iso = next(
    (c for c in world_gpd.columns
     if "iso" in c.lower() and "a3" in c.lower() and "eh" not in c.lower()),
    None
)
if _raw_iso and _raw_iso != "iso_a3":
    world_gpd = world_gpd.rename(columns={_raw_iso: "iso_a3"})

# ── 3. Merge UNICEF data onto world geometries ────────────────────────────
zinc_pd = zinc_avg.to_pandas()
world_merged = world_gpd.merge(
    zinc_pd[["alpha_3_code", "zinc_avg"]],
    left_on="iso_a3",
    right_on="alpha_3_code",
    how="left"
)

# ── 4. Plot ───────────────────────────────────────────────────────────────
(
    ggplot(world_merged)
    + geom_map(aes(fill="zinc_avg"), color="white", size=0.1)
    + scale_fill_gradientn(
        colors=["#ffffcc", "#fed976", "#feb24c", "#fd8d3c", "#f03b20", "#bd0026"],
        na_value="#d0d0d0",
        name="Coverage (%)"
    )
    + coord_equal()
    + theme_void()
    + theme(
        plot_title=element_text(size=13, weight="bold", ha="center"),
        plot_subtitle=element_text(size=9, color="#555555", ha="center"),
        legend_title=element_text(size=9),
        legend_text=element_text(size=8),
        plot_background=element_rect(fill="#f8f9fa", color=None)
    )
    + labs(
        title="Global Zinc Treatment Coverage for Childhood Diarrhoea",
        subtitle=(
            "Average % of children under 5 with diarrhoea who received zinc"
            " | Grey = No data"
        )
    )
)


# ── Helper: returns the most recent Total-sex value per country ───────────
def get_latest_by_country(
    df: pl.DataFrame, value_col: str = "obs_value"
) -> pl.DataFrame:
    """Extracts the most recent 'Total' observation per country with Polars."""
    return (
        df
        .filter(pl.col("sex") == "Total")
        .group_by(["country", "alpha_3_code"])
        .agg([
            pl.col(value_col).sort_by("time_period").last().alias("latest_value"),
            pl.col("time_period").max().alias("year")
        ])
    )

dep_latest = get_latest_by_country(ind1)

# ── Convert from proportion (0–1) to percentage (0–100) ──────────────────
top_dep = (
    dep_latest
    .filter(pl.col("latest_value") > 0)
    .with_columns((pl.col("latest_value") * 100).alias("dep_pct"))
    .sort("dep_pct", descending=True)
    .head(20)
    .to_pandas()
)

# ── Horizontal bar chart ──────────────────────────────────────────────────
(
    ggplot(top_dep, aes(x="reorder(country, dep_pct)", y="dep_pct"))
    + geom_bar(stat="identity", fill="#c0392b", alpha=0.85, width=0.72)
    + coord_flip()
    + theme_minimal()
    + theme(
        axis_text_y=element_text(size=9),
        axis_text_x=element_text(size=9),
        plot_title=element_text(size=13, weight="bold"),
        plot_subtitle=element_text(size=9, color="#555555"),
        panel_grid_major_y=element_blank(),
        panel_grid_minor=element_blank()
    )
    + labs(
        title="Countries with Highest Rates of Severe Child Deprivation",
        subtitle=(
            "% of children suffering 5+ simultaneous deprivations"
            " | Most recent available year"
        ),
        x=None,
        y="Deprivation Rate (%)"
    )
)


# ── 1. Most recent zinc rate per country ──────────────────────────────────
zinc_latest = (
    get_latest_by_country(ind2, "obs_value")
    .rename({"latest_value": "zinc_rate"})
)

# ── 2. Average GDP & life expectancy per country (2010–2022) ──────────────
gdp_avg = (
    meta
    .filter(pl.col("year").is_between(2010, 2022))
    .filter(pl.col("gdp_per_capita").is_not_null())
    .group_by("alpha_3_code")
    .agg([
        pl.col("gdp_per_capita").mean().alias("avg_gdp"),
        pl.col("life_expectancy").mean().alias("avg_life_exp")
    ])
)

# ── 3. Inner join on alpha_3_code ─────────────────────────────────────────
scatter_data = (
    zinc_latest
    .join(gdp_avg, on="alpha_3_code", how="inner")
    .filter(pl.col("avg_gdp").is_not_null())
    .filter(pl.col("zinc_rate").is_not_null())
    .to_pandas()
)

# ── 4. Scatterplot with OLS regression line ───────────────────────────────
(
    ggplot(scatter_data, aes(x="avg_gdp", y="zinc_rate"))
    + geom_point(aes(size="avg_life_exp"), color="#2980b9", alpha=0.55)
    + geom_smooth(method="lm", color="#e74c3c", fill="#fadbd8", alpha=0.25)
    + scale_x_log10()
    + theme_minimal()
    + theme(
        plot_title=element_text(size=13, weight="bold"),
        plot_subtitle=element_text(size=9, color="#555555"),
        axis_title=element_text(size=10),
        legend_title=element_text(size=9)
    )
    + labs(
        title="Wealthier Countries Provide Better Child Healthcare",
        subtitle=(
            "GDP per capita (log scale) vs zinc coverage"
            " | Point size = life expectancy"
        ),
        x="GDP per Capita, USD — Log Scale",
        y="Zinc Treatment Coverage (%)",
        size="Life Expectancy\n(years)"
    )
)


# ── Countries representing diverse regional & economic trajectories ────────
selected_countries = [
    "Bangladesh", "Afghanistan", "Nigeria", "India",
    "Kenya", "Cambodia", "Bolivia", "Niger"
]

# ── Filter, select, and sort time-series data ─────────────────────────────
ts_data = (
    ind2
    .filter(pl.col("sex") == "Total")
    .filter(pl.col("country").is_in(selected_countries))
    .select(["country", "time_period", "obs_value"])
    .sort(["country", "time_period"])
    .to_pandas()
)

# ── One distinct colour per country ───────────────────────────────────────
country_palette = {
    "Bangladesh": "#1a9641",
    "Afghanistan": "#d7191c",
    "Nigeria":     "#fdae61",
    "India":       "#2c7bb6",
    "Kenya":       "#756bb1",
    "Cambodia":    "#31a354",
    "Bolivia":     "#f46d43",
    "Niger":       "#a50026"
}

# ── Time-series chart ─────────────────────────────────────────────────────
(
    ggplot(ts_data, aes(x="time_period", y="obs_value",
                        color="country", group="country"))
    + geom_line(size=1.2, alpha=0.85)
    + geom_point(size=2.5, alpha=0.9)
    + scale_color_manual(values=country_palette)
    + scale_x_continuous(breaks=list(range(2005, 2026, 2)))
    + theme_minimal()
    + theme(
        plot_title=element_text(size=13, weight="bold"),
        plot_subtitle=element_text(size=9, color="#555555"),
        axis_text_x=element_text(angle=45, ha="right"),
        legend_title=element_blank(),
        panel_grid_minor=element_blank()
    )
    + labs(
        title="Zinc Treatment Coverage Trends Over Time",
        subtitle=(
            "% of children under 5 with diarrhoea receiving zinc"
            " | Selected countries, 2005–2023"
        ),
        x="Year",
        y="Zinc Treatment Coverage (%)"
    )
)
